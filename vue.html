<template>
    <div>
        <!-- <input type="text" v-model="userId"> -->
        <input type="text" :value="userId" @input="userId = $event.target.value">
        <button @click="myFunction">click</button>      <!-- @click = addEventListener("click", ()=>{}) -->
        <button @click="changeData">change</button>
        <br>
        <input type="text" v-model="num1"> +      <!-- 문자열로 인식 -->
        <input type="text" v-model="num2"> =
        <span>{{ num1 + num2 }}</span>
        <br>
        <input type="text" v-model.number="num3"> +       <!-- .number을 붙이면 숫자로 인식 -->
        <input type="text" v-model.number="num4"> =
        <span>{{ num3 + num4 }}</span>
    </div>
</template>
Composition 
<script setup>     // 컴포지션 방식에서는 setup 필수
    import { ref, onMounted } from 'vue';

    let userId = ref('abc');
    let num1 = ref(0);
    let num2 = ref(0);
    let num3 = ref(0);
    let num4 = ref(0);

    function myFunction () {
        console.log(userId.value);
    }
    
    function changeData () {
        userId.value = 'kyubi'
    }

    onMounted(() => {
        ...     // template 내 html이 모두 다 불러와졌을 때 실행
    })
</script>
Options 
<script>
    export default {
        data () {
            return {    // 모든 데이터는 여기에서 선언
                userId: 'abcd',
                num1: 0,
                num2: 0,
                num3: 0,
                num4: 0,
            }
        },
        methods: {      // 모든 함수는 여기에서 선언
            myFunction () {
                console.log(this.userId)
            },
            changeData () {
                this.userId = 'kyubi'
            }
        }
    }
</script>
Options 방식과는 달리 Composition 방식에서는 필요한 부분만 불러와서 사용
반응형 데이터 만들기
ref | 범위 제한이 없음, 데이터 변경 시 .value를 붙여서 변경
reactive | obj,arr,Map,set 등 범위가 제한적, 데이터 변경 시 .value 붙일 필요 없음
