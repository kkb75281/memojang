level 0 | 짝수 홀수 개수

정수가 담긴 리스트 num_list가 주어질 때, num_list의 원소 중 짝수와 홀수의 개수를 담은 배열을 return 하도록 solution 함수를 완성해보세요.

1 ≤ num_list의 길이 ≤ 100
0 ≤ num_list의 원소 ≤ 1,000
myAnswer 
function solution(num_list) {
    var answer = [];
    let a = [];
    let b = [];

    for (let num of num_list) {
        if (num % 2 === 0) {
            a.push(num);
        } else {
            b.push(num);
        }
    }

    answer[0] = a.length;
    answer[1] = b.length;

    return answer;
}
others 
function solution(num_list) {
    var answer = [0,0];

    for(let a of num_list){
        answer[a%2] += 1
    }

    return answer;
}
