함수는 정의하기 전에 호출 가능하지만, 클래스는 반드시 위에 먼저 선언 후 호출 가능

# class 선언 (기본형식)

// 보통 class명은 대문자로 시작
class DateCard {
    // 여기서도 마찬가지로 let 안씀
    // height;
    // width;

    // 매개변수 연결
    constructor(year, month, day, where, who, why) {
        this.year = year;
        this.month = month;
        this.day = day;
        this.where = where;
        this.who = who;
        this.why = why;
    }

    // 함수 (재활용가능) : function today() {} or let today = ()=>{} 이런 형식은 아님
    today() {
        console.log(`Hello! Today is ${this.day} ${this.month}, ${this.year}`);

        // 함수 안에서는 일반 자바스크립트처럼 사용 (let 선언 가능)
        let data = {
            year: this.year,
            month: `${this.month}`,
            day: this.day
        }

        return data;
    }
}

[ promise ]

자바스크립트는 코드를 순차적으로 읽어 나감. 하지만 promise를 사용하면 특정 코드가 완료될 때까지 기다리지 않고 바로 다음 코드로 넘어가서 실행함.

promise를 꼭 함수로 표현할 필요는 없음. 다만 let a = new Promise() 식으로 표현하면 웹브라우저를 실행할때마다 계속해서 호출이 됨

ex. 게임 아이템 구매 : 클릭했을 때 구매과정이 이루어져야 함

# promise

// 01. 실행할 때마다 계속 호출
let p = new Promise(resolved => {
    console.log('hello');
    resolved('Finished!')
})

// then을 통해 결과를 전송
p.then(r => {
    console.log(r);
})



// 02. 함수를 통해 필요할때만 promise를 호출할 수 있음
let promiseFunction = () => new Promise(resolved => {
    console.log('hello promise function!');
    resolved('Resolved!')
});

promiseFunction().then(pf => {
    console.log(pf);
})

let v1 = 1;
let v2 = 10;

let promise = new Promise((then, err) => {
    let a = v1 + v2;

    if (a == 2) {
        then('success');
    } else {
        // then('failed');
        err('failed');
    }
})

////////////////////////////////////////// err

promise.then((message) => {
    console.log('This is in the then ' + message)
}, err => {
    console.log('-err-');
    console.log('This is in the then ' + err)
});
// -err-
// This is in the then failed

//////////////////////////////////////// catch

promise().then((message) => {
    console.log('This is in the then ' + message)
}).catch(err=>{
    console.log('-catch-', err);
    console.log('This is in the catch ' + err)
});
// -catch-
// This is in the catch failed

# try ... catch

보통 스크립트에서 에러가 발생하면 실행이 중단되고 콘솔에 에러가 뜨지만 try ... catch 문을 사용하면 스크립트가 죽지 않고 에러에 해당하는 코드를 실행 할 수 있음.

실행 가능한 코드(유효한 자바스크립트 코드)에만 실행, 중괄호 짝이 안 맞는 것처럼 코드가 문법적으로 잘못된 경우에는 실행되지 않음

.catch(f)는 promise.then(null, f)과 동일하게 작동

의도된 오류를 잡는데 쓰임(ex. 같은 아이디를 적었을 경우 > 알림창을 띄운다) 
try {
    // 코드
    // 1. 코드 실행
    // 2. 에러가 없다면 try 안 코드 실행되고 catch 건너뜀
} catch (err) {
    // 에러 핸들링
    // 3. 에러가 있다면 try 코드 실행 중단, 경우에 따라 해당되는 catch 문 실행
} finally {
    // 항상 실행
}

function fetchItems() {
    return new Promise(function(resolve, reject) {
        var items = [1,2,3];
        resolve(items);
    });
}
    
async function logItems() {
    var resultItems = await fetchItems();
    console.log(resultItems); // [1,2,3]
}

function fetchUser() {
    var url = 'https://jsonplaceholder.typicode.com/users/1'
    return fetch(url).then(function (response) {
        return response.json();
    });
}

function fetchTodo() {
    var url = 'https://jsonplaceholder.typicode.com/todos/1';
    return fetch(url).then(function (response) {
        return response.json();
    });
}

async function logTodoTitle() {
    var user = await fetchUser();
    if (user.id === 1) {
        var todo = await fetchTodo();
        console.log(todo.title);
    }
}
